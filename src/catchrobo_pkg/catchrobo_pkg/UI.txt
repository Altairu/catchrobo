<body>
    <div class="container">
        <h1>Robot Control Panel</h1>
        <div id="buttons">
            <button class="button" onclick="toggleAction(0)">Action 0</button>
            <button class="button" onclick="toggleAction(1)">Action 1</button>
            <button class="button" onclick="toggleAction(2)">Action 2</button>
            <button class="button" onclick="toggleAction(3)">Action 3</button>
            <button class="button" onclick="toggleAction(4)">Action 4</button>
        </div>
        <div class="status" id="status">Status: Disconnected</div>
        <div class="info" id="currentAction">Current Action: None</div>
        <div class="info" id="specialAction">Special Action: None</div>
        <div class="info" id="signalStrength">Signal Strength: Unknown</div>
        <div class="info" id="actionDescription">Action Description: None</div>
    </div>

    <script>
        let websocket;
        let currentAction = null;
        let specialAction = null;
        const actionDescriptions = [
            "Description for Action 0",
            "Description for Action 1",
            "Description for Action 2",
            "Description for Action 3",
            "Description for Action 4"
        ];

        function connectWebSocket() {
            websocket = new WebSocket("ws://192.168.11.47:8080/ws");

            websocket.onopen = function () {
                document.getElementById('status').innerText = "Status: Connected";
                document.getElementById('status').className = "status connected";
            };

            websocket.onclose = function () {
                document.getElementById('status').innerText = "Status: Disconnected";
                document.getElementById('status').className = "status disconnected";
            };

            websocket.onmessage = function (event) {
                const data = event.data.split(",");
                document.getElementById('currentAction').innerText = `Current Action: ${data[0]}`;
                document.getElementById('specialAction').innerText = `Special Action: ${data[1]}`;
                document.getElementById('signalStrength').innerText = `Signal Strength: ${data[2]}`;
                document.getElementById('actionDescription').innerText = `Action Description: ${actionDescriptions[data[0]] || "None"}`;
            };
        }

        function toggleAction(actionNumber) {
            if (currentAction === actionNumber) {
                currentAction = null;
            } else {
                currentAction = actionNumber;
            }

            if (websocket && websocket.readyState === WebSocket.OPEN) {
                websocket.send(currentAction !== null ? currentAction.toString() : "None");
            }
        }

        connectWebSocket();
    </script>
